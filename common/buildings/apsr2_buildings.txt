apsr2_building_part1 = {
	base_buildtime = @apsr2part1cost_time
	icon = "apsr2_building_part1"
	base_cap_amount = 1
	
	is_capped_by_modifier = no

	category = resource
	
#	can_demolish = no
#	can_be_ruined = no
	
	potential = {
		AND = {
			has_modifier = "apsr2_planet_modifier"
			NOT = {
				OR = {
					has_building = apsr2_building_part2
					has_building = apsr2_building_part3
					has_building = apsr2_building_part4
					has_building = apsr2_building_part5
					has_building = apsr2_building_part1_fake
					has_building = apsr2_building_part2_fake
					has_building = apsr2_building_part3_fake
					has_building = apsr2_building_part4_fake
					has_building = apsr2_building_part5_fake
				}
			}
		}
	}
			
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_apsr2_miner_hive_add = @apsr2part1job
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_apsr2_miner_machine_add = @apsr2part1job
		}
	}	
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		modifier = {
			job_apsr2_miner_normal_add = @apsr2part1job
		}
	}
	
	resources = {
		category = planet_apsr_buildings
		cost = {
			minerals = @apsr2part1cost_minerals
		}
		upkeep = {
			energy = @apsr2part1upkeep_energy
			minerals = @apsr2part1upkeep_minerals
		}
	}
	
	upgrades = {
		"apsr2_building_part2"
	}	
	
	prerequisites = {
		"tech_apsr2_building_part1"
	}
	
	ai_weight = {
		weight = 50000
	}
}
