apsr13_building_part1 = {
	base_buildtime = @apsr13part1cost_time
	icon = apsr13_building_part1
	base_cap_amount = 1
	
	is_capped_by_modifier = no

	category = trade
	
#	can_demolish = no
#	can_be_ruined = no
	
	potential = {
		AND = {
			has_modifier = "apsr13_planet_modifier"
			NOT = {
				OR = {
					has_building = apsr13_building_part2
					has_building = apsr13_building_part3
					has_building = apsr13_building_part1_fake
					has_building = apsr13_building_part2_fake
					has_building = apsr13_building_part3_fake
				}
			}
		}
	}

			
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_apsr13_trader_hive_add = @apsr13part1job
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_apsr13_trader_machine_add = @apsr13part1job
		}
	}	
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		modifier = {
			job_apsr13_trader_normal_add = @apsr13part1job
		}
	}
	
	resources = {
		category = planet_apsr_buildings
		cost = {
			minerals = @apsr13part1cost_minerals
		}
		upkeep = {
			energy = @apsr13part1upkeep_energy
			minerals = @apsr13part1upkeep_minerals
		}
	}
	
	upgrades = {
		"apsr13_building_part2"
	}	
	
	prerequisites = {
		"tech_apsr13_building_part1"
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	ai_weight = {
		weight = 50000
	}
}
