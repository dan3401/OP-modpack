apsr3_building_part1 = {
	base_buildtime = @apsr3part1cost_time
	icon = apsr3_building_part1
	base_cap_amount = 1
	
	is_capped_by_modifier = no

	category = research
	
#	can_demolish = no
#	can_be_ruined = no
	
	potential = {
		AND = {
			has_modifier = "apsr3_planet_modifier"
			NOT = {
				OR = {
					has_building = apsr3_building_part2
					has_building = apsr3_building_part3
					has_building = apsr3_building_part4
					has_building = apsr3_building_part1_fake
					has_building = apsr3_building_part2_fake
					has_building = apsr3_building_part3_fake
				}
			}
		}
	}

			
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_hive_empire = yes }
		}
		modifier = {
			job_apsr3_scientist_hive_add = @apsr3part1job
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_machine_empire = yes }
		}
		modifier = {
			job_apsr3_scientist_machine_add = @apsr3part1job
		}
	}	
		
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_gestalt = no
			}
		}
		modifier = {
			job_apsr3_scientist_normal_add = @apsr3part1job
		}
	}
	
	resources = {
		category = planet_apsr_buildings
		cost = {
			minerals = @apsr3part1cost_minerals
		}
		upkeep = {
			energy = @apsr3part1upkeep_energy
			minerals = @apsr3part1upkeep_minerals
		}
	}
	
	upgrades = {
		"apsr3_building_part2"
	}	
	
	prerequisites = {
		"tech_apsr3_building_part1"
	}
	
	allow = {
		has_upgraded_capital = yes
	}
	
	ai_weight = {
		weight = 50000
	}
}
