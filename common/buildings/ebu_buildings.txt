#Farming Buildings

building_hydroponics_farm = {
	base_buildtime = @b1_time

	category = resource

	potential = {
		exists = owner
		owner = {
			OR = {
				country_uses_food = yes
				is_ai = no
			}
		}
		NOT = { is_planet_class = pc_machine }
		NOT = { has_modifier = resort_colony }
	}

	allow = {
		buildings_simple_allow = yes
	}

	destroy_trigger = {
		exists = owner
		OR = {
			owner = {
				is_ai = yes
				country_uses_food = no
			}
			planet = {
				NOT = { is_planet_class = pc_habitat }
				free_building_slots = 0
				owner = {
					is_ai = yes
					has_monthly_income = {
						resource = food
						value > 100
					}
				}
			}
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_regular_empire = yes }
		}
		modifier = {
			job_farmer_add = 2
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_origin = origin_void_dwellers }
		}
		modifier = {
			job_farmer_add = 1
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_agri_drone_add = 2
		}
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			energy = 2
		}
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_agri_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_farmer_effect_desc
	}

	prerequisites = {
		"tech_hydroponics"
	}
	
	upgrades = {
		"eb_hydrofarm2"
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 10
			NOR = {
				is_planet_class = pc_habitat
				is_planet_class = pc_city
				is_planet_class = pc_relic
			}
			OR = {
				free_jobs < 3
				buildings_unemployed = yes
			}
			OR = {
				num_free_districts = { type = district_farming value = 0 }
				free_district_slots = 0
			}
			owner = {
				has_monthly_income = {
					resource = food
					value < 0
				}
			}
		}
		modifier = {
			weight = 20
			OR = {
				free_housing > 3
				owner = { is_gestalt = no }
			}
			OR = {
				free_jobs < 3
				buildings_unemployed = yes
			}
			is_planet_class = pc_habitat
			OR = {
				owner = {
					has_monthly_income = {
						resource = food
						value < 50
					}
				}
				has_building = building_hydroponics_farm
			}
		}
		modifier = {
			factor = 0
			owner = {
				has_monthly_income = {
					resource = food
					value > 100
				}
			}
		}
		modifier = {
			factor = 0
			owner = {
				country_uses_food = no
			}
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
