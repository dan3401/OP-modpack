create_lost_empire_reinforcements = {
		create_fleet = {
			effect = {
				set_owner = prev.owner
				while = {
					count = 2
					create_lost_empire_battlecruiser = yes
				}
				while = {
					count = 8
					create_lost_empire_escort = yes
				}
			set_location = this
		}
	}		
}

create_random_lost_empire_fleet = {
	create_fleet = {
		effect = {
			set_owner = prev.owner
			random_list = {
				25 = {
					create_lost_empire_escort = yes
				}
				25 = {
					create_lost_empire_battlecruiser = yes
				}
				25 = {
					while = {
						count = 3
						create_lost_empire_escort = yes
					}
				}
				25 = {
					while = {
						count = 3
						create_lost_empire_escort = yes
					}
					create_lost_empire_battlecruiser = yes
				}
			}
			set_location = root
		}
	}
}
