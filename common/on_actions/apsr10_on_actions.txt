# обозначения:
# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# работа с тупой логикой ботов (70х)
# фейк-замена (80х)
# сообщение при замене фейк-зданий на настоящие (81х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.10001 # сообщение - этап 1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.10001 # сообщение - этап 1 построен
	}
}

on_survey = {
	events = {
		apsr.10011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.10013 # сообщение при захвате планеты
		apsr.10801 # замена зданий на фейки при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.10013 # сообщение при получении планеты
		apsr.10801 # замена зданий на фейки при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.10013 # сообщение при колонизации планеты
		apsr.10801 # замена зданий на фейки при колонизации планеты (скрытый)
	}
}

on_tech_increased = {
	events = { 
		apsr.10802 # замена фейков на здания при изучении технологии (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.10014 # отображение найденного ресурса, не показаного раньше
		apsr.10901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}
