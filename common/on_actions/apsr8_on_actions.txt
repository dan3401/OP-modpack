# постройка и улучшение здания (00х)
# нахождение ресурса (01х)
# работа с тупой логикой ботов (70х)
# фейк-замена (80х)
# важные системные ивенты (90х)
# прочие ивенты (6хх)



on_building_complete = {
	events = {
		apsr.8001 # сообщение - этап 1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.8001 # сообщение - этап 1 построен
	}
}

on_building_upgraded = {
	events = {
		apsr.8002 # сообщение - этап 2 построен
		apsr.8003 # сообщение - этап 3 построен
	}
}

on_survey = {
	events = {
		apsr.8011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.8013 # сообщение при захвате планеты
		apsr.8801 # замена зданий на фейки при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.8013 # сообщение при получении планеты
		apsr.8801 # замена зданий на фейки при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.8013 # сообщение при колонизации планеты
		apsr.8801 # замена зданий на фейки при колонизации планеты (скрытый)
	}
}

on_tech_increased = {
	events = { 
		apsr.8802 # замена фейков на здания при изучении технологии (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.8014 # отображение найденного ресурса, не показаного раньше
		apsr.8901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}
