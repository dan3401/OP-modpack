# нахождение ресурса (01х)

# прочие ивенты (6хх)

on_building_complete = {
	events = {
		apsr.24001 # сообщение - электростанция построена
		apsr.24002 # сообщение - комплекс талисман построен
		apsr.24031 # запуск отсчета до истощения электростанции (скрытый)
	}
}

on_building_replaced = {
	events = {
		apsr.24001 # сообщение - электростанция построена
		apsr.24002 # сообщение - комплекс талисман построен
		apsr.24031 # запуск отсчета до истощения электростанции (скрытый)
	}
}

on_building_upgraded = {
	events = {
		apsr.24031 # запуск отсчета до истощения электростанции (скрытый)
	}
}

on_survey = {
	events = {
		apsr.24011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.24013 # сообщение при захвате планеты
		apsr.24801 # замена зданий на фейки при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.24013 # сообщение при получении планеты
		apsr.24801 # замена зданий на фейки при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.24013 # сообщение при колонизации планеты
		apsr.24801 # замена зданий на фейки при колонизации планеты (скрытый)
	}
}

on_tech_increased = {
	events = { 
		apsr.24802 # замена фейков на здания при изучении технологии (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.24014 # отображение найденного ресурса, не показаного раньше
		apsr.24901 # проверка наличия комбинации модификатор-депозит (скрытый)
		apsr.24021 # смена производимого Талисманом ресурса
	}
}
