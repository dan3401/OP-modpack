on_building_complete = {
	events = {
		apsr.13001 # сообщение - этап 1 построен
	}
}

on_building_replaced = {
	events = {
		apsr.13001 # сообщение - этап 1 построен
	}
}

on_building_upgraded = {
	events = {
		apsr.13002 # сообщение - этап 2 построен
		apsr.13003 # сообщение - этап 3 построен
	}
}

on_survey = {
	events = {
		apsr.13011 # сообщение при изучении научным кораблем
	}
}

on_planet_conquer = {
	events = {
		apsr.13013 # сообщение при захвате планеты
		apsr.13801 # замена зданий на фейки при захвате планеты (скрытый)
	}
}

on_planet_transfer = {
	events = {
		apsr.13013 # сообщение при получении планеты
		apsr.13801 # замена зданий на фейки при передаче планеты (скрытый)
	}
}

on_colonization_started = {
	events = {
		apsr.13013 # сообщение при колонизации планеты
		apsr.13801 # замена зданий на фейки при колонизации планеты (скрытый)
	}
}

on_tech_increased = {
	events = { 
		apsr.13802 # замена фейков на здания при изучении технологии (скрытый)
	}
}

on_yearly_pulse = {
	events = {
		apsr.13014 # отображение найденного ресурса, не показаного раньше
		apsr.13901 # проверка наличия комбинации модификатор-депозит (скрытый)
	}
}
