district_teleipopolis_normal = {
	base_buildtime = 400
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		OR = { 
			is_planet_class = pc_eternity
			is_planet_class = pc_eternal_ringworld_habitable
		}
		
		from = { 
			exists = owner
			owner = { has_government = gov_eternal_continuum }
		}
	}
	
	potential = {
		OR = { 
			is_planet_class = pc_eternity
			is_planet_class = pc_eternal_ringworld_habitable
		}
		exists = owner
		owner = { has_government = gov_eternal_continuum }
	}
		
	resources = {
		category = planet_districts
		cost = {
			minerals = 5000
			ambrosia = 1000
			virtue = 50
		}
		upkeep = {
			ambrosia = 10
			energy = 5
		}
	}
	
	conversion_ratio = 1
	convert_to = {
		district_city
		district_nexus
		district_hive
		district_arcology_housing
	}

	planet_modifier = {
		planet_housing_add = 12
		planet_amenities_add = 5
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { has_government = gov_eternal_continuum }
		}
		modifier = {
			
		}
	}
	
	ai_weight = {
		weight = 2
		modifier = {
			factor = 2
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 0
				}
			}
		}
		modifier = {
			factor = 5
			free_housing < 0
		}

		# Let's not go full spam on this.
		modifier = {
			factor = 0
			free_housing > 2
		}
	}

	ai_resource_production = {
		consumer_goods = 1
		trigger = {
			always = yes
		}
	}
}
