		modifier = {

		}
		modifier = {
			factor = 1.5
			OR = {

					area = engineering

				}

					area = engineering

				}
			}		
		}
		modifier = {
			factor = 5
			has_origin = apsr_origin_unique_capital
		}
	}
	
	ai_weight = {
		factor = 150
	}	
}
