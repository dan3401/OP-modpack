#patriarchy military
country_event = {
	id = eternal_leader_events.1
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		has_policy_flag = eternal_gender_patriarchy_military
		from = {
			OR = {
				leader_class = admiral
				leader_class = general
			}
			gender = female
		}
	}

	immediate = {
		from = {
			set_is_female = no
			set_name = random
		}
	}
}
