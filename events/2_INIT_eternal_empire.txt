country_event = {
	id = setuptw.3
	title = OK
	desc = OK
	hide_window = yes
	is_triggered_only = yes	
	fire_only_once = yes
	
	
	trigger = {
		any_country = {
			if = {
			limit = {
					OR = {
					has_government = "gov_eternal_continuum"
					has_government = "gov_ancient_ai"
					}
					is_ai = yes
				}
			}
		}
	}

	immediate = {
		every_country = {
			if = {
				limit = {
					OR = {
					has_government = "gov_eternal_continuum"
					has_government = "gov_ancient_ai"
					}
					is_ai = yes	
				}
			add_modifier = {
				modifier = "all_years_fe"
				days = 60000
			}
			add_modifier = {
				modifier = "first_years_fe"
				days = 2000
			}
			add_modifier = {
				modifier = "immortal"
				days = -1
			}
			add_modifier = {
				modifier = "first_years_fallen"
				days = 3600
			}
			
			
			country_event =  { id = setuptw.4 }
			country_event =  { id = setuptw.13 }

			}
		}
		
	}
}
