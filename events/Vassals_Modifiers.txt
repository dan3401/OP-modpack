namespace = twvasall

country_event = {
	id = twvasall.1
	title = OK
	desc = OK
	hide_window = yes
	is_triggered_only = yes
	
	immediate = {
		every_country = {	
			if = {
				limit = {
					is_subject_type = eternal_vassal
				}
				
				add_modifier = {
					modifier = "vasal"
					days = 360
				}
			}
		}
	}
}
