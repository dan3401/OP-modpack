namespace = ex_game_start
### Empire init - add technologies and resources
country_event = {
	id = ex_game_start.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = { 																	# Psionic Heritage Civic Parameters, credit to Alfray Stryke for showing me how to implement this
			limit = { 
				has_civic = civic_psionic_heritage 
			} 
			give_technology = { message = no tech = tech_psionic_theory }		# Gives empire Psionic Theory for free
		}
			}
}
